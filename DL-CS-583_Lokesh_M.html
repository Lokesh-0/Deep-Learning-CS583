# machine learning neural network model:
def mc_model(input_size=dataframe.shape[1], output_size=len(set(label))):
    model = Sequential()
    model.add(layers.Dropout(0.5, input_shape=(input_size, )))
    model.add(layers.Dense(10, kernel_initializer='uniform', bias_initializer='zeros'))
    model.add(layers.LayerNormalization(axis=1))
    model.add(layers.Activation('relu'))
    model.add(layers.Dropout(0.5))
    model.add(layers.Dense(output_size, 
                          kernel_initializer='uniform', bias_initializer='uniform'))
    model.add(layers.LayerNormalization(axis=1))
    model.add(layers.Activation('softmax'))
    return model
